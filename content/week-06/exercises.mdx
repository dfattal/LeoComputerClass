```python
def AND(a, b):
    return a & b

def OR(a, b):
    return a | b

def NOT(a):
    return 1 - a

def XOR(a, b):
    return OR(AND(a, NOT(b)), AND(NOT(a), b))

def mux(a, b, s):
    return OR(AND(NOT(s), a), AND(s, b))

def get_bit(x, i):
    return (x >> i) & 1
```
